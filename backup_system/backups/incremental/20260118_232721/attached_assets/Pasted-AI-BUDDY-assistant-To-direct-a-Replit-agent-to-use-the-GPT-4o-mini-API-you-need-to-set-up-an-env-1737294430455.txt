# Define the API endpoint and your API key
API_URL = "https://api.gpt-4o-mini.com/v1/complete"  # Replace with the actual API endpoint

# Function to call the GPT-4o-mini API
def gpt_4o_mini_query(prompt):
    headers = {
        "Authorization": f"Bearer {API_KEY}",
        "Content-Type": "application/json"
    }
    
    data = {
        "prompt": prompt,
        "max_tokens": 100,  # You can adjust this parameter
        "temperature": 0.7  # You can adjust the creativity of the response
    }
    
    response = requests.post(API_URL, headers=headers, json=data)
    
    if response.status_code == 200:
        return response.json()  # Process the JSON response
    else:
        return {"error": response.json()}

# Example usage
